    <script>

        const books = [

            "The Young Team",

            "The Storied Life of A.J. Fikry", 

            "Venomous Lumpsucker",

            "Starter Villain",

            "The Edinburgh Skating Club",

            "Prophet Song",

            "Foundry",

            "Lessons in Chemistry",

            "House of Suns",

            "God of the Woods"

        ];


        const members = ["Ben", "Ed", "Mesfin", "Jamie", "Dave", "George"];

        let currentMember = null;

        let rankings = {};

        let draggedElement = null;


        function init() {

            updateSubmissionCount();

            

            document.getElementById('memberSelect').addEventListener('change', function(e) {

                currentMember = e.target.value;

                if (currentMember) {

                    showRankingSection();

                    populateBookList();

                }

            });

            

            document.getElementById('submitBtn').addEventListener('click', submitRankings);

            document.getElementById('showResultsBtn').addEventListener('click', showResults);

            document.getElementById('exportBtn').addEventListener('click', exportResults);

        }


        function showRankingSection() {

            document.getElementById('rankingSection').classList.add('active');

        }


        function populateBookList() {

            const bookList = document.getElementById('bookList');

            bookList.innerHTML = '';

            

            const existingRanking = rankings[currentMember] || [...books];

            

            existingRanking.forEach((book, index) => {

                const bookItem = createBookItem(book, index + 1);

                bookList.appendChild(bookItem);

            });

        }


        function createBookItem(title, rank) {

            const item = document.createElement('div');

            item.className = 'book-item';

            item.draggable = true;

            item.innerHTML = `

                <div class="book-rank">${rank}</div>

                <div class="book-title">${title}</div>

                <div style="color: #999;">⋮⋮</div>

            `;

            

            item.addEventListener('dragstart', handleDragStart);

            item.addEventListener('dragover', handleDragOver);

            item.addEventListener('drop', handleDrop);

            item.addEventListener('dragend', handleDragEnd);

            

            return item;

        }


        function handleDragStart(e) {

            draggedElement = e.target;

            e.target.classList.add('dragging');

        }


        function handleDragOver(e) {

            e.preventDefault();

        }


        function handleDrop(e) {

            e.preventDefault();

            if (draggedElement !== e.target && e.target.classList.contains('book-item')) {

                const bookList = document.getElementById('bookList');

                const allItems = [...bookList.children];

                const draggedIndex = allItems.indexOf(draggedElement);

                const targetIndex = allItems.indexOf(e.target);

                

                if (draggedIndex < targetIndex) {

                    bookList.insertBefore(draggedElement, e.target.nextSibling);

                } else {

                    bookList.insertBefore(draggedElement, e.target);

                }

                

                updateRankNumbers();

            }

        }


        function handleDragEnd(e) {

            e.target.classList.remove('dragging');

            draggedElement = null;

        }


        function updateRankNumbers() {

            const bookItems = document.querySelectorAll('.book-item');

            bookItems.forEach((item, index) => {

                const rankElement = item.querySelector('.book-rank');

                rankElement.textContent = index + 1;

            });

        }


        function submitRankings() {

            if (!currentMember) return;

            

            const bookItems = document.querySelectorAll('.book-item');

            const memberRanking = [];

            

            bookItems.forEach(item => {

                const title = item.querySelector('.book-title').textContent;

                memberRanking.push(title);

            });

            

            rankings[currentMember] = memberRanking;

            

            const status = document.getElementById('submitStatus');

            status.className = 'status success';

            status.textContent = `✓ Rankings submitted successfully for ${currentMember}!`;

            status.style.display = 'block';

            

            document.getElementById('submitBtn').disabled = true;

            document.getElementById('submitBtn').textContent = 'Rankings Submitted ✓';

            

            updateSubmissionCount();

        }


        function updateSubmissionCount() {

            const count = Object.keys(rankings).length;

            document.getElementById('submissionCount').textContent = count;

        }


        function showResults() {

            const resultsSection = document.getElementById('resultsSection');

            const resultsContent = document.getElementById('resultsContent');

            

            resultsContent.innerHTML = '';

            

            members.forEach(member => {

                if (rankings[member]) {

                    const memberDiv = document.createElement('div');

                    memberDiv.className = 'member-results';

                    memberDiv.innerHTML = `

                        <h3>${member}'s Rankings</h3>

                        <ol class="result-list">

                            ${rankings[member].map((book, index) => `

                                <li><div class="result-rank">${index + 1}</div>${book}</li>

                            `).join('')}

                        </ol>

                    `;

                    resultsContent.appendChild(memberDiv);

                }

            });

            

            resultsSection.classList.add('active');

        }


        function exportResults() {

            let csv = 'Member,' + books.join(',') + '\n';

            

            members.forEach(member => {

                if (rankings[member]) {

                    const row = [member];

                    books.forEach(book => {

                        const position = rankings[member].indexOf(book) + 1;

                        row.push(position);

                    });

                    csv += row.join(',') + '\n';

                }

            });

            

            const blob = new Blob([csv], { type: 'text/csv' });

            const url = window.URL.createObjectURL(blob);

            const a = document.createElement('a');

            a.href = url;

            a.download = 'book_club_rankings.csv';

            a.click();

            window.URL.revokeObjectURL(url);

        }


        init();

    </script>
